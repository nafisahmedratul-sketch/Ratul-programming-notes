#example of scanf and printf
```c
int main() {
    int age;
    float height;
    float waight;
    printf("Enter your age: ");
    scanf("%d", &age);
    printf("Enter your height: ");
    scanf("%f", &height);
    printf("Enter your waight: ");
    scanf("%f", &waight);
    printf("You are %d years old.\n", age);
    printf("Your height is =%f\n", height);
    printf("Your weight is.%f\n", waight);
    return 0;
}
```
